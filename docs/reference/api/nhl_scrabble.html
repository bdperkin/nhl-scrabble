158    def _calculate_backoff_delay(self, attempt: int, retry_after: int | None = None) -> float:
159        """Calculate backoff delay with exponential backoff and jitter.
160
161        Args:
162            attempt: Current attempt number (0-indexed)
163            retry_after: Optional Retry-After header value from 429 response
164
165        Returns:
166            Delay in seconds with jitter applied
167
168        Examples:
169            >>> client = NHLApiClient()
170            >>> client._calculate_backoff_delay(0)  # First retry
171            0.75  # ~1.0 * (2.0 ** 0) with ±25% jitter
172            >>> client._calculate_backoff_delay(3)  # Fourth retry
173            6.5   # ~8.0 * (2.0 ** 3) with ±25% jitter, capped at max_backoff
174        """
175        if retry_after is not None:
176            # Respect Retry-After header from API (429 responses)
177            return min(float(retry_after), self.max_backoff)
178
179        # Exponential backoff: base_delay * (backoff_factor ** attempt)
180        base_delay = 1.0
181        delay = min(base_delay * (self.backoff_factor**attempt), self.max_backoff)
182
183        # Add jitter: randomize ±25% to prevent thundering herd
184        # Safe: Using random for jitter, not cryptography
185        jitter = delay * 0.25
186        delay = delay + random.uniform(-jitter, jitter)  # noqa: S311
187
188        return max(0, delay)
